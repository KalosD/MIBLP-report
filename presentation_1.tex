\section{双层规划问题}

\begin{frame}
	\frametitle{双层规划问题的标准形式} 
	\begin{equation}
		\begin{aligned}
			\min_{x,y}& F(x,y)  \\
			&G(x,y)\leq0 \\
			&y\in\arg\operatorname*{min}_{y^{\prime}}\{f(x,y^{\prime}):g(x,y^{\prime})\leq0\},
		\end{aligned}
	\end{equation}
	\begin{itemize}
		\item $x$为上层变量, $y$为下层变量;
		\item $F(x,y)$和$f(x,y)$分别为上层和下层的目标函数;
		\item $G(x,y)$和$g(x,y)$分别为上层和下层的约束条件.
	\end{itemize}
\end{frame}
